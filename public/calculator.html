from tkinter import *
FONT2 = ("Angsana New", 20)

def numberClicked(n):
    global answerStatus
    textOnScreen = screen.get()
    
    if (len(textOnScreen)==0) and (n==0):
        return
    else:
        screen.insert(len(textOnScreen), n)

    if answerStatus==True:
        screen.delete(0, END)
        screen.insert(len(textOnScreen), n)
        answerStatus = False
    return

def equal():
    global answerStatus
    textOnScreen = screen.get()
    
    calculationResult = eval(textOnScreen)
    # 'eval':This function is used to evaluates the string expression directly
    screen.delete(0, END)
    screen.insert(len(textOnScreen), calculationResult)
    answerStatus = True
    
def clearResult():
    global answerStatus
    screen.delete(0, END)
    #Tkinter Text widget has delete(first, last=None) method to delete the characters at the index of first, or within the range of (first, last) from the text box.
    answerStatus = False

def back():
    global answerStatus
    #The rstrip() method removes trailing characters from the right side of a string by default. You can optionally specify the characters to remove within the parentheses.
    textOnScreen = screen.get()
    previousText = textOnScreen[:-1]
    screen.delete(0, END)
    screen.insert(len(textOnScreen), previousText)
    answerStatus = False

def percentage():
    global answerStatus
    textOnScreen = screen.get()
    screen.insert(len(textOnScreen), "* (1/100)")
    calculationResult = eval(screen.get())
    # 'eval':This function is used to evaluates the string expression directly
    screen.delete(0, END)
    screen.insert(len(screen.get()), calculationResult)
    answerStatus = True

def operator(operate):
    global answerStatus
    textOnScreen = screen.get()
    screen.insert(len(textOnScreen), operate)
    answerStatus = False

appScreen = Tk()
appScreen.title("Calculator")
appScreen.resizable(0,0)
screen = Entry(appScreen,font=FONT2,width=37)
screen.grid(row=0,column=0,columnspan=4,rowspan=2)
answerStatus = False
